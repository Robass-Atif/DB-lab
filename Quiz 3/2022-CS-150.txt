SELECT e.first_name + ' ' + e.last_name AS employee_name,d.department_name AS department_name,CASE WHEN DATENAME(WEEKDAY, DATEADD(MONTH, 6, e.hire_date)) = 'Friday' THEN 'Friday, the ' + CONVERT(VARCHAR(2), DAY(DATEADD(MONTH, 6, e.hire_date))) + ' of ' +DATENAME(MONTH, DATEADD(MONTH, 6, e.hire_date)) + ', ' +CONVERT(VARCHAR(4), YEAR(DATEADD(MONTH, 6, e.hire_date))) ELSE  'Friday, the ' + CONVERT(VARCHAR(2), DAY(DATEADD(WEEKDAY, 6 - DATEPART(WEEKDAY, DATEADD(MONTH, 6, e.hire_date)), DATEADD(MONTH, 6, e.hire_date)))) + ' of ' + DATENAME(MONTH, DATEADD(MONTH, 6, e.hire_date)) + ', ' +CONVERT(VARCHAR(4), YEAR(DATEADD(MONTH, 6, e.hire_date)))  END AS salary_review_date FROM HR.dbo.employees AS e INNER JOIN  HR.dbo.departments AS d ON e.department_id = d.department_id
